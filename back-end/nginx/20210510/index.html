# 安装依赖项

yum install -y openssl-devel pcre pcre-devel zlib zlib-devel

# 下载

wget http://nginx.org/download/nginx-1.18.0.tar.gz

# 解压

tar zxvf nginx-1.18.0.tar.gz
cd nginx-1.18.0

# 安装

./configure --prefix=/opt/module/nginx
make && make install
